Datentypen in JavaScript: Strings

        // String-Variablen, ganz trivial, als Literale oder als Objekt:
        var string1 = "erster";
        var string2 = "erster";
        var string3 = new String("erster");
        console.log("typeof string1/string3");
        console.log(typeof string1);
        console.log(typeof string3);
        var result;

        // String-Vergleich a) Vergleich auf Wert, ungeachtet des Typs
        result = (string1 == string2);
        console.log('string1 == string2');
        console.log(result);
        console.log('string1 == string3');
        console.log(string1 == string3);

        // String-Vergleich a) Vergleich auf Wert- und Typ-Gleichheit
        result = (string1 === string2);
        console.log('string1 === string2');
        console.log(result);
        console.log('string1 === string3');
        console.log(string1 === string3);

        // String-Vergleich c) str.localeCompare
        // Wichtig: das ursprüngliche String-Literal (primitiver string!)
        // wird hier implizit in ein String-Objekt gecastet!
        result = (string1.localeCompare(string2));
        console.log('string1.localeCompare(string2)');
        console.log(result);
        // string1 ist hier "wieder" ein string - kein object:
        console.log(typeof string1);

        // Prüfung auf "kommt vorher"
        result = (string1 < "abc");
        console.log('string1 < "abc"');
        console.log(result);
        result = (string1 > "abc");
        console.log('string1 > "abc"');
        console.log(result);
