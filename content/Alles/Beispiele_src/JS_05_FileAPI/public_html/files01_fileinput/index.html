      function check(){
        var fileIn = document.getElementById('mein_file');
        var info   = document.getElementById('info');

        var files = fileIn.files;
        var l = files.length;
        info.innerHTML += '' + l + ' Dateien 
';
        for (i = 0; i < l; i++) {
          var f = files[i];
          info.innerHTML += 'Datei: ' + f.name + '
';
          info.innerHTML += '... Größe: ' + f.size + ' Bytes
';
          info.innerHTML += '... MIME: '  + f.type+ '
';
        }
      }
