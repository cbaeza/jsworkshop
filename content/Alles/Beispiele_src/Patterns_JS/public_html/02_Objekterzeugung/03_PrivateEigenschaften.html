Private Eigenschaft


      /*
       * Private Eigenschaft:
       */
      function Objekt(){
        // Achtung! Das, was "privat" sein soll, kann nicht mehr
        // als Objekt-Eigenschaft definiert werden - wäre ja immer sichtbar.
        // Daher: Implementierung als lokale Variable
        var id = 123456;

        /*
         * "Getter-Funktion"
         */
        this.getID = function(){
          return id;
        };
        /*
         * "Setter-Funktion"
         */
        this.setID = function(vId){
          id = vId;
        };
      };

      var myObj = new Objekt();
      console.log(myObj.getID());
      myObj.setID("Neuer Wert!");
      console.log(myObj.getID());
