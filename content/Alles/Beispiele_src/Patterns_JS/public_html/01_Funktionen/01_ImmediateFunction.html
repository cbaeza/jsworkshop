Immediate Function & Immediate Method


      /*
       * Immediate Function
       * - Motivation: Scope! globale Variablen minimieren!
       */
      (function(){
        var msg = 'Immediate Function';
        console.log(msg);
      })();

      /*
       * Immediate Function mit Parameter
       * - z.B. window-Objekt wird benötigt,
       *   soll aber nicht explizit referenziert
       */
      (function(o){
        o.msg = 'Immediate Function';
        console.log(o.innerHeight);
      }(this));

      /*
       * Immediate Function im Objekt
       */
      var obj = {
        // Umschließende Klammern nur zur syntaktischen
        // Verdeutlichung:
        eigenschaft: (function(){
          return 'ergebnis einer aufwendigen Berechnung';
        }())
      };
      console.log(JSON.stringify(obj));

      /*
       * Immediate Method
       */
      ({
        maxWidth: 400,
        maxHeight: 300,
        toolMethode: function(){
          return Math.sqrt(this.maxHeight * this.maxWidth);
        },
        init:function(){
          console.log(this.toolMethode());
        }
      }.init());
