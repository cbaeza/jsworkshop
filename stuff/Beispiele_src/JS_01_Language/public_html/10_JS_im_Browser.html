      // Zugriff auf das Browser-Fernster über das in der JavaScript-Umgebung
      // vorhandene window-Objekt
      window.onload = handleOnLoad;


      // Funktions-Deklaration - wird dank "Hoisting" schon zu Beginn des
      // Scripts initialisiert.
      function handleOnLoad() {
        console.log('onload OK');

        // Zugriff auf das HTML-Dokument über das in der JavaScript-Umgebung
        // vorhandene document-Objekt
        var container = document.getElementById("container");

        // Manipulation des Document Object Models (DOM)
        var p = document.createElement("P");
        p.appendChild(document.createTextNode("Dies ist dynamisch generierter Inhalt."));
        container.appendChild(p);

        // Achtung: CamelCase-Syntax bei Zugriff auf CSS-Eigenschaften
        p.style.color = "#336699";
        p.style.fontFamily = 'serif';
        p.style.backgroundColor = '#eeeeee';


        // Iteration durch die DOM-Elemente
        var nodesList = container.childNodes;
        var l = nodesList.length;
        for (var i = 0; i < l; i++) {
          var element = console.log(nodesList[i]);
        }
      }
