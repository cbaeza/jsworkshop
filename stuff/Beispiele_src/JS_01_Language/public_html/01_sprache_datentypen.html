Datentypen in JavaScript


      // String-Literal
      var a = '123456';
      var a1 = new String("ein String"); // ist redundant!
      // Verwendung von Escape-Sequenzen in Strings
      // a = '123\n456';
      // alert(a);

      // Numerisches Literal
      var b = 123456;
      // hex-Syntax
      b = 0x123456;
      // float
      b = 0.123456;

      // Verarbeitung von Strings in der Praxis
      // String-Concat
      var c = a + 9;

      // ebenfalls String-Concat!
      c = 9 + a;

      // ebenfalls String-Concat!
      c = b + a;

      // Erzwingen von Typumwandlungen:
      var d = '100';
      // Zuweisung mit unärem +-Operator:
      // erzwingt die Umwandlung zu number
      var e = +d;

      // Umwandlung in String erzwingen:
      var f = 1000;
      // f soll nun z.B. für String-Ausgabe konkatiniert werden
      var g = '' + (f * e) + " ist mein Ergebnis";
      console.log(g);

      console.log(e);
      console.log(typeof e);
      console.log(e + 23);

      // ACHTUNG: "Type Juggling"!
      // Kontextabhängig kann der Interpreter den Typ unserer Variablen ändern!
