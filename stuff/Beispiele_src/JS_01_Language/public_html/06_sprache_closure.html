Closures in JavaScript


      /*
       * Zum Beriff Closure:
       * - "Funktion mit Referenz auf äußeren Kontext"
       * - Allgemein: "Funktion mit eigenem Scope + Code"
       */

       // Funktion mit äußerem Kontext
       function aeussereFunktion(){
         var eineVariableInFunc = "Wert";
         var innereFunc = function (){
           console.log("ich arbeite mit eineVariableInFunc: " + eineVariableInFunc);
         };
         return innereFunc;
       };


      function handleOnLoad(){
        // Aufruf der äußeren Funktion:
        var f = aeussereFunktion();
        // Rückgabewert ist ein Funktionsobjekt,
        // das Zugriff auf die Variable im Scope der äußeren Funktion hat
        f();
      }
