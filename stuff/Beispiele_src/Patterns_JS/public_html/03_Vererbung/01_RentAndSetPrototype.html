Rent-And-Set-Prototype


      /*
       * Rent-And-Set-Prototype
       * (ACHTUNG!)
       */
      function Parent(name){
        this.name = name || 'Adam';
      }

      Parent.prototype.say = function(){
        return this.name;
      }
//      var p = new Parent();
//      console.log(p.say());


      function Child(name){
        // Aufrufen der Parent-Funktion mit Scope "this"
        // und den eigenen Funktions-Argumenten
        Parent.apply(this, arguments);
      }
      Child.prototype = new Parent();

      // Anwendung
      var c = new Child("Eva");
      console.log(c.say());

      // WT?!!!!:
      // Eigenschaft name löschen:
      delete c.name;
      console.log(c.say());
