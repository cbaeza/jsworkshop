Vererbung durch Kopie


      /*
       * Kopier-Funktion
       */
      function extend(parent, child) {
        var i;

        child = child || {};
        for (i in parent) {
          // Achtung: Kopie oder Referenz?!
          // => Primitive Typen (Kopie) oder Objekt-Typen (Referenz)!
          child[i] = parent[i];
        }

        return child;
      };

      var myParent = {
        name: "Adam",
        say: function(){
          return "Ich bin " + this.name;
        }
      };

      console.log(myParent.say());

      var myChild = extend(myParent);

      console.log(myChild.say());
